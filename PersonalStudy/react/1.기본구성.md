## 간단한 버튼 만들기
### simple1.html
* `react-root`는 js에서 사용할 DOM 요소임
* `<script src="simple1.js"/></script>` 사용할 js를 추가함
```html
<html>
    <body>
        <div id="react-root"></div> 
        <script src="react.development.js"></script>
        <script src="react-dom.development.js"/></script>
        <script src="simple1.js"/></script>
    </body>
</html>
```
### simple1.js* 
```js
// 좋아요 버튼 만들기 
// LikeButton을 React.Component로 상속받아서 생성
class LikeButton extends React.Component{
    constructor(props){
        super(props);
        this.state = { liked: false };
    }
    render(){
        const text = this.state.liked ? '좋아요 취소' : '좋아요';
        return React.createElement(
            'button',
            {onClick: () => this.setState({ liked: true })},
            text,
        );
    }
}
// react-root DOM 요소에 LikeButton 추가 작업
ReactDOM.render(React.createElement(LikeButton), document.querySelector('#react-root'));
```
